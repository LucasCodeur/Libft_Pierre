25char *ft_strnstr(const char *big, const char *little, size_t len)
26{
27 size_t little_size;
28 size_t count;
29
30 count = 0;
31 little_size = ft_strlen(little);
32 if (little_size == 0 || little == big)
33 return ((char *)&big[0]);
34 if (len == 0 || *big == '\0' || ft_strlen(big) < little_size
35 || len < little_size)
36 return (NULL);
37 while (big[count] != '\0' && count <= (len - little_size))
38 {
39 if (ft_strncmp(&big[count], little, little_size) == 0)
40 return ((char *)&big[count]);
41 count++;
42 }
43 if (count > (len - little_size))
44 return (NULL);
45 return ((char *)&big[0]);
46}
