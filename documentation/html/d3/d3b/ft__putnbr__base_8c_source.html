23ssize_t ft_putnbr_base(int nbr, char *base)
24{
25 int base_nbr;
26 long int nb;
27 ssize_t temp;
28
29 temp = 0;
30 nb = nbr;
31 base_nbr = ft_check_base_putnbr(base);
32 if (base_nbr < 1)
33 return (0);
34 if (nb < 0)
35 {
36 nb = -nb;
37 temp += write(1, "-", 1);
38 }
39 if (nb >= base_nbr)
40 {
41 temp += ft_putnbr_base(nb / base_nbr, base);
42 temp += ft_putnbr_base(nb % base_nbr, base);
43 }
44 else
45 temp += write(1, &base[nb], 1);
46 return (temp);
47}
