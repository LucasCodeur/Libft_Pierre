29char **ft_split(char const *s, char c)
30{
31 char **ptr_ptr;

33 size_t nbr_cuts;
34
35 count = 0;
36 nbr_cuts = str_count_char(s, c);
37 ptr_ptr = malloc(sizeof(char *) * (nbr_cuts + 1));
38 if (ptr_ptr == NULL)
39 return (NULL);
40 while (nbr_cuts-- > 0)
41 {
42 while (*s == c && *s != '\0')
43 s++;
44 ptr_ptr[count] = ft_substr(s, 0, ft_strchr_end(s, c) - s);
45 if (ptr_ptr[count++] == NULL)
46 {
47 ft_free_malloc(ptr_ptr, count - 1);

49 }

51 }
